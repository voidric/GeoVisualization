const GeoVisSystem = {
    // 系统初始化入口
    init() {
        this.initRenderWindow(); 
        this.initSceneAndInteraction(); 
        this.initGUI();
        this.switchMode('terrain');
        this.animate(); 
    },

    // UI交互界面初始化
    initGUI() {
        this.gui = new GUI({ title: '三维地质可视系统', width: 320 });
        
        const sysConfig = { mode: 'terrain' };
        this.gui.add(sysConfig, 'mode', {
            '地形可视化': 'terrain',
            '地震可视化': 'seismic'
        }).onChange(this.switchMode.bind(this)).name("切换系统");

        this.gui.addColor({ color : '#222222' }, 'color')
            .name('场景背景')
            .onChange(c => this.scene.background.set(c));
    },

    // 模块切换与资源调度
    switchMode(mode) {
        if (this.currentContainer) {
            this.scene.remove(this.currentContainer);
            this.currentContainer = null;
        }

        if (mode === 'terrain') this.initTerrainModule();
        else this.initSeismicModule();
    },

    // 三维渲染窗口的创建与初始化
    initRenderWindow() {
        this.renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });
        this.renderer.setSize(window.innerWidth, window.innerHeight);
        this.renderer.setPixelRatio(window.devicePixelRatio);
        document.body.appendChild(this.renderer.domElement);

        this.camera = new THREE.PerspectiveCamera(60, window.innerWidth / window.innerHeight, 0.1, 50000);
        this.camera.position.set(1500, 1500, 1500);
    },

    // 三维场景管理与基础交互功能
    initSceneAndInteraction() {
        this.scene = new THREE.Scene();
        this.scene.background = new THREE.Color(0x222222);

        // 灯光系统
        this.scene.add(new THREE.AmbientLight(0xffffff, 0.6));
        const dirLight = new THREE.DirectionalLight(0xffffff, 0.8);
        dirLight.position.set(500, 1500, 500);
        this.scene.add(dirLight);

        // 辅助显示
        this.scene.add(new THREE.GridHelper(5000, 50));
        this.scene.add(new THREE.AxesHelper(500));

        // 交互控制器
        this.controls = new OrbitControls(this.camera, this.renderer.domElement);
        this.controls.enableDamping = true;

        window.addEventListener('resize', () => {
            this.camera.aspect = window.innerWidth / window.innerHeight;
            this.camera.updateProjectionMatrix();
            this.renderer.setSize(window.innerWidth, window.innerHeight);
        });
    },

    // 渲染循环
    animate() {
        requestAnimationFrame(this.animate.bind(this));
        this.controls.update();
        this.renderer.render(this.scene, this.camera);
    }
};
